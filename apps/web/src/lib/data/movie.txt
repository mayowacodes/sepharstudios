import { getXataClient } from '../xata'; // Assuming your Xata client is in '../xata.ts'
import type { MediaItem } from '../types/media'; // Adjust the path if necessary

const xata = getXataClient();

export async function fetchXataMovies(): Promise<MediaItem[]> {
  try {
    const movies = await xata.db.movies.getMany();
    return movies.map((movie) => ({
      id: `m-${movie.id}`, // Prefix to distinguish from local data
      title: movie.title || 'Untitled Movie',
      description: movie.overview || '',
      thumbnail: movie.poster_url || '',
      trailerUrl: undefined, // No direct mapping
      link: `/watch/movie/${movie.id}`,
      rating: movie.vote_average ? movie.vote_average.toString() : undefined,
      duration: undefined, // No direct mapping
      quality: undefined, // No direct mapping
      year: movie.release_date ? new Date(movie.release_date).getFullYear().toString() : undefined,
      genre: movie.genres || [],
      bibleReference: undefined, // No direct mapping
      ageRating: undefined, // No direct mapping
      backdropUrl: movie.backdrop_url || undefined,
      genreIds: movie.genre_ids || undefined,
      originalLanguage: movie.original_language || undefined,
      popularity: movie.popularity || undefined,
      releaseDate: movie.release_date || undefined,
      voteCount: movie.vote_count || undefined,
      keywords: movie.keywords || undefined,
      keywordIds: movie.keyword_ids || undefined,
      keywordsList: (movie as any).keywords_list || undefined, // Assuming this might exist
    }));
  } catch (error) {
    console.error('Error fetching movies from Xata:', error);
    return [];
  }
}
