<script lang="ts">
  import { onMount, createEventDispatcher } from 'svelte';
  import { fade } from 'svelte/transition';
  import { Play, Info, Volume2, VolumeX, BookOpen } from '@lucide/svelte';
  import type { MediaItem } from '../types/media';

  const dispatch = createEventDispatcher();

  export let items: MediaItem[] = [];
  let currentIndex = 0;
  let videoRef: HTMLVideoElement | null = null;
  let isMuted = true;
  let isPlaying = false;

  onMount(() => {
    const interval = setInterval(() => {
      if (!isPlaying && items.length > 0) {
        currentIndex = (currentIndex + 1) % items.length;
      }
    }, 8000);
    
    return () => clearInterval(interval);
  });

  function toggleMute() {
    if (videoRef) {
      videoRef.muted = !videoRef.muted;
      isMuted = videoRef.muted;
    }
  }

  function handlePlay() {
    dispatch('play', items[currentIndex]);
  }

  function handleMoreInfo() {
    dispatch('moreInfo', items[currentIndex]);
  }
</script>

<div class="relative h-[80vh] overflow-hidden">
  {#if items.length > 0}
    {@const item = items[currentIndex]}
    <div class="absolute inset-0" transition:fade={{ duration: 800 }}>
      {#if isPlaying && item.trailerUrl}
        <video
          bind:this={videoRef}
          src={item.trailerUrl}
          class="w-full h-full object-cover"
          autoplay
          loop
          muted={isMuted}
        />
      {:else}
        <img
          src={item.thumbnail || "/placeholder.svg"}
          alt={item.title}
          class="w-full h-full object-cover"
          loading="lazy"
        />
      {/if}

      <div class="absolute inset-0 bg-gradient-to-t from-black via-black/50 to-transparent" />

      <div class="absolute inset-0 flex items-center">
        <div class="container mx-auto px-4">
          <div class="max-w-2xl space-y-6">
            <h1 class="text-4xl md:text-5xl font-bold text-white">{item.title}</h1>

            <p class="text-lg md:text-xl text-white/90 line-clamp-3">{item.description}</p>

            <div class="flex flex-wrap items-center gap-3 text-sm text-white/80">
              {#if item.year}<span>{item.year}</span>{/if}
              {#if item.rating}
                <span class="px-1.5 py-0.5 bg-[#FF5E0E] rounded text-white">
                  {item.rating}
                </span>
              {/if}
              {#if item.duration}<span>{item.duration}</span>{/if}
              {#if item.ageRating}
                <span class="px-1.5 py-0.5 border border-white/50 rounded">
                  {item.ageRating}
                </span>
              {/if}
            </div>

            <div class="flex flex-wrap items-center gap-4">
              <button
                class="px-6 py-2 md:px-8 md:py-3 bg-[#FF5E0E] text-white rounded-lg hover:bg-[#FFBF00] transition-colors flex items-center gap-2"
                on:click={handlePlay}
              >
                <Play class="h-5 w-5" />
                Play
              </button>
              
              <button
                class="px-6 py-2 md:px-8 md:py-3 bg-white/20 text-white rounded-lg hover:bg-white/30 transition-colors flex items-center gap-2"
                on:click={handleMoreInfo}
              >
                <Info class="h-5 w-5" />
                More Info
              </button>
              
              {#if item.bibleReference}
                <button
                  class="px-4 py-2 bg-[#2A2A2A] text-white rounded-lg hover:bg-[#3A3A3A] transition-colors flex items-center gap-2"
                  on:click={() => dispatch('showScripture', item.bibleReference)}
                >
                  <BookOpen class="h-4 w-4" />
                  Scripture
                </button>
              {/if}
              
              {#if isPlaying}
                <button
                  class="p-2 md:p-3 rounded-full bg-black/50 text-white hover:bg-white/20 transition-colors"
                  on:click={toggleMute}
                  aria-label={isMuted ? 'Unmute' : 'Mute'}
                >
                  {#if isMuted}
                    <VolumeX class="h-5 w-5" />
                  {:else}
                    <Volume2 class="h-5 w-5" />
                  {/if}
                </button>
              {/if}
            </div>
          </div>
        </div>
      </div>
    </div>
  {:else}
    <!-- Fallback when no items -->
    <div class="absolute inset-0 bg-gray-900 flex items-center justify-center">
      <p class="text-white/80 text-xl">Loading featured content...</p>
    </div>
  {/if}
</div>
